#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/ipc.h>
#include <sys/shm.h>
#include <sys/wait.h>

#define SHM_SIZE 1024

int main()
{
    key_t key = 5678;

    int shmid = shmget(key, SHM_SIZE, IPC_CREAT | 0666);

    if (shmid == -1)
    {
        perror("shmget");
        exit(1);
    }

    char *memory = (char *)shmat(shmid, NULL, 0);

    if (memory == (char *)-1)
    {
        perror("shmat");
        exit(1);
    }

    pid_t pid = fork();

    if (pid < 0)
    {
        perror("fork");
        exit(1);
    }

    if (pid == 0)
    {
        strcpy(memory, "Operating Systems Practical 4: Shared Memory Communication");
        printf("Child Process:\n");
        printf("Message written to shared memory:\n%s\n", memory);

        shmdt(memory);
        exit(0);
    }
    else
    {
        wait(NULL);

        printf("\nParent Process:\n");
        printf("Message received from shared memory:\n%s\n", memory);

        shmdt(memory);
        shmctl(shmid, IPC_RMID, NULL);
    }

    return 0;
}
